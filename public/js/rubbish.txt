    closeOtherModals(chatMembersModal);

function renderMembers() {

    membersList.innerHTML = "";

    roomData.members.forEach(member => {

        const div = document.createElement("div");
        div.className = "member";

        div.dataset.id = member.user;

        div.innerHTML = `
            <div class="icon">
                <span class="material-symbols-outlined">person</span>
            </div>

            <div class="memberInfo">
                <div class="memberName">
                    ${member.user.name}
                </div>

                <div class="memberRole">
                    ${member.role}
                </div>
            </div>

            <div class="memberActions">

            </div>
        `;

        membersList.append(div);
        if (canManage(member)) {

            const actions = div.querySelector(".memberActions");
            actions.innerHTML = `
        <span class="material-symbols-outlined promoteBtn">
            arrow_upward
        </span>

        <span class="material-symbols-outlined removeBtn">
            person_remove
        </span>
    `;

        }

    });

}


function canManage(member) {
    console.log(member)
    const me = roomData.members.find(m => m.user._id === myId);

    if (!me) return false;

    if (me.role === "owner")
        return member.user._id !== myId;

    if (me.role === "admin")
        return member.role === "member";

    return false;

}
